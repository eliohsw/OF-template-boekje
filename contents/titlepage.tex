% ============ Title Settings ====================================

\makeatletter
\pretitle{%
  \vspace*{6cm}%   Similar vertical space from the top
  % \centering
  \fontsize{40}{48}\selectfont\BebasNeue\bfseries        % Font & size for the main title
}
\posttitle{%
  \par\vspace{0.1cm}%  spacing before subtitle
  \huge\sffamily\sbseries       % Font & size for the subtitle
  \@subtitle\par                % Insert the subtitle (we'll define \@subtitle below)
  \vspace{2cm}%                 % spacing before author
}

\newcommand{\subtitle}[1]{\gdef\@subtitle{#1}} % helper command to store subtitle text
\newcommand{\@subtitle}{}                      % default empty

\preauthor{%
  % \centering
  \Large\sffamily              % Font & size for author
}
\postauthor{%
  \par\vspace{0.25cm}
  \large{\@idnumber}\par       % We'll define \@idnumber for that "B123456" line
  \vspace{3.5cm}               % Then next space before date
}

\newcommand{\idnumber}[1]{\gdef\@idnumber{#1}}
\newcommand{\@idnumber}{}   % default empty

\predate{%
  % \centering
  \Large
}
\postdate{%
  \par\vspace{0.25cm}
  \Large{\@placename}\par    % We'll define \@placename
  \vfill
}

\newcommand{\placename}[1]{\gdef\@placename{#1}}
\newcommand{\@placename}{}   % default empty
\makeatother
